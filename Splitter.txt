.animated-arc-border {
  position: absolute;
  inset: 0;
  border-radius: 50%;
  /* This conic-gradient creates the arc effect */
  background: conic-gradient(
    from 0deg,
    #ffc107, /* Yellowish-orange start color */
    #ff5722, /* Orangish-red end color */
    transparent 180deg /* Make the rest of the circle transparent */
  );
  /* This is crucial for making the gradient part of the border */
  /* It creates an inner transparent circle, leaving only the gradient arc visible */
  -webkit-mask: radial-gradient(farthest-side, transparent calc(100% - 4px), black calc(100% - 4px + 1px));
  mask: radial-gradient(farthest-side, transparent calc(100% - 4px), black calc(100% - 4px + 1px));
}
