// TextInput.jsx

import React from 'react';
import './TextInput.css';

const TextInput = ({ inputText, onInputChange, onKeyPress, placeholder, ...props }) => {
  const handleInputHeight = (e) => {
    e.target.style.height = 'auto';
    e.target.style.height = `${e.target.scrollHeight}px`;
  };

  return (
    <textarea
-      className="flex-grow resize-none p-3 border border-gray-300 rounded-lg shadow-sm focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent text-sm max-h-40 overflow-y-auto align-middle"
+      className="flex-grow resize-none py-3 px-6 border border-gray-400 rounded-full shadow-sm focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent text-sm max-h-40 overflow-y-auto align-middle placeholder-gray-500 pl-10"
      value={inputText}
      onChange={onInputChange}
      onKeyDown={onKeyPress}
      onInput={handleInputHeight}
      placeholder={placeholder}
      {...props}
    />
  );
};

export default TextInput;
