// Add this list of greetings
const greetings = [
  'Hi there!',
  'Hello again!',
  'Welcome back!',
  'How can I help?',
  "What's on your mind?"
];

// Use useState to manage the current greeting
const [currentGreeting, setCurrentGreeting] = useState('');

// Use useEffect to set a random greeting when the component loads
useEffect(() => {
  const randomIndex = Math.floor(Math.random() * greetings.length);
  setCurrentGreeting(greetings[randomIndex]);
}, []); // The empty array [] ensures this runs only once
