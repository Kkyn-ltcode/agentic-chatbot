import re

text_list = [
    "สวัสดีHelloสวัสดี",
    "สวัสดี Helloสวัสดี",
    "สวัสดี 123สวัสดี",
    "สวัสดี123 สวัสดี"
]

# Pattern: match Thai, Latin, or digits (ignore spaces and punctuation)
pattern = r'[ก-๙]+|[A-Za-z]+|\d+'

for text in text_list:
    tokens = re.findall(pattern, text)
    print(tokens)
